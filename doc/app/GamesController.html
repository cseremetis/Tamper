def simulator
    unless params[:player] == nil
        @game_id = fetch
        
        @all = Player.where(:game_id => @game_id)
        @name = params[:player][:name]
        
        if check(@name, @all)
            @player = Player.new
            @player.game_id = @game_id
            @player.name = params[:player][:name]
            @player.age = params[:player][:age].to_i
            @player.skill = params[:skill].to_i
            @player.improvise = params[:improvise].to_i
            @player.combat = params[:combat].to_i
            @player.sociability = params[:sociability].to_i
            @player.kindness = params[:kindness].to_i
            @player.emotions = params[:emotions].to_i
            @player.intelligence = params[:intelligence].to_i
            @player.save
            
            @alert = false
        else
            @alert = true
        end
        
        #initialize new variable with added players
        @every_player = Player.where(:game_id => @game_id)
    end
end
